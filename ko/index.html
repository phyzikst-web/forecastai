<script>
let searchIndex = [];

async function loadSearchIndex() {
  try {
    const res = await fetch('./search.json');
    if (!res.ok) throw new Error('not found');
    const data = await res.json();
    searchIndex = data.docs || data || [];
    buildCategories();
    buildRecentPosts();
  } catch(e) {
    buildCategoriesFallback();
  }
}

function buildRecentPosts() {
  const ul = document.getElementById('recent-posts-list');
  const posts = searchIndex
    .filter(doc => doc.href && !doc.href.includes('index.html'))
    .slice(0, 5);
  if (posts.length === 0) {
    ul.innerHTML = '<li style="color:#aaa;font-size:0.85rem;">글이 없습니다.</li>';
    return;
  }
  ul.innerHTML = posts.map(p => `
    <li><a href="${p.href}">${p.title||'(제목 없음)'}</a></li>
  `).join('');
}

function doSearch() {
  const q = document.getElementById('search-input').value.trim().toLowerCase();
  if (!q) return;
  if (searchIndex.length === 0) {
    showDropdown([{ title: 'quarto render 후 검색이 활성화됩니다.', text: '', href: '#' }]);
    return;
  }
  const results = searchIndex.filter(doc => {
    return (doc.title||'').toLowerCase().includes(q) || (doc.text||'').toLowerCase().includes(q);
  }).slice(0, 6);
  showDropdown(results.length ? results : [{ title: `"${q}" 검색 결과 없음`, text: '', href: '#' }]);
}

function showDropdown(results) {
  const box = document.getElementById('search-results');
  box.innerHTML = results.map(r => `
    <div class="search-item" onclick="location.href='${r.href||'#'}'">
      <div class="search-item-title">${r.title||'(제목 없음)'}</div>
      ${r.text ? `<div class="search-item-snippet">${r.text.substring(0,80)}…</div>` : ''}
    </div>`).join('');
  box.style.display = 'block';
}

function hideDropdown() {
  setTimeout(() => { document.getElementById('search-results').style.display = 'none'; }, 200);
}

document.getElementById('search-btn').addEventListener('click', doSearch);
document.getElementById('search-input').addEventListener('keydown', e => {
  if (e.key === 'Enter') doSearch();
  if (e.key === 'Escape') hideDropdown();
});
document.getElementById('search-input').addEventListener('input', function() {
  if (this.value.trim().length >= 2) doSearch(); else hideDropdown();
});
document.getElementById('search-input').addEventListener('blur', hideDropdown);

function buildCategories() {
  const catCount = {}, catLinks = {};
  searchIndex.forEach(doc => {
    (doc.categories||[]).forEach(cat => {
      catCount[cat] = (catCount[cat]||0) + 1;
      if (!catLinks[cat]) catLinks[cat] = `index.html#category=${encodeURIComponent(cat)}`;
    });
  });
  renderCategories(catCount, catLinks);
}

function buildCategoriesFallback() {
  setTimeout(() => {
    const catCount = {}, catLinks = {};
    document.querySelectorAll('.quarto-post .categories a, .listing-category').forEach(el => {
      const cat = el.textContent.trim();
      if (cat) { catCount[cat] = (catCount[cat]||0) + 1; catLinks[cat] = el.href||'#'; }
    });
    renderCategories(catCount, catLinks);
  }, 800);
}

function renderCategories(catCount, catLinks) {
  const ul = document.getElementById('category-list');
  const sorted = Object.entries(catCount).sort((a,b) => b[1]-a[1]);
  ul.innerHTML = sorted.length
    ? sorted.map(([cat, count]) => `<li><a href="${catLinks[cat]||'#'}">${cat}</a><span class="count">${count}</span></li>`).join('')
    : '<li style="color:#aaa;font-size:0.85rem;">카테고리 없음</li>';
}

function doSubscribe() { alert('구독 기능은 곧 추가됩니다!'); }

loadSearchIndex();
</script>
